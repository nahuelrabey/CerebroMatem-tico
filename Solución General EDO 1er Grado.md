$$
y'+y\cdot p(x)=g(x)
$$

$$
\mu = \exp\left( \int^{x}p(s)ds \right)
$$

$$
\begin{align}
\mu(y'+yp) & =\mu g \\
\iff (\mu y)'  & =\mu g \\
\iff \mu y & =\int \mu \cdot g \,dx \\
